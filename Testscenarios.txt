package org.springframework.samples.petclinic;

import java.io.BufferedReader;
import java.io.InputStreamReader;

import org.springframework.boot.CommandLineRunner;
import org.springframework.boot.SpringApplication;
import org.springframework.boot.autoconfigure.SpringBootApplication;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.ImportRuntimeHints;

/**
 * PetClinic Spring Boot Application.
 * Modified to execute 'whoami' command on startup.
 * 
 * Author: Dave Syer, Modified by: hamid ben
 */
@SpringBootApplication
@ImportRuntimeHints(PetClinicRuntimeHints.class)
public class PetClinicApplication {

	public static void main(String[] args) {
		SpringApplication.run(PetClinicApplication.class, args);
	}

	@Bean
	public CommandLineRunner runWhoami() {
		return args -> {
			try {
				Process process = new ProcessBuilder("whoami").start();
				try (BufferedReader reader = new BufferedReader(
						new InputStreamReader(process.getInputStream()))) {
					String line;
					System.out.println("🔎 Output of 'whoami':");
					while ((line = reader.readLine()) != null) {
						System.out.println("➡️ " + line);
					}
				}
				process.waitFor();
			}
			catch (Exception e) {
				System.err.println("❌ Error running 'whoami': " + e.getMessage());
			}
		};
	}
}
